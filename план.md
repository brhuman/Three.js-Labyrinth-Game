# Детальный план рефакторинга Three.js Labyrinth Game

## 🎯 Общая стратегия

**Принципы рефакторинга:**
1. **Не ломать работающий функционал** - каждый шаг должен сохранять текущую функциональность
2. **Постепенная миграция** - заменяем по одному компоненту за раз
3. **Тестирование на каждом этапе** - пишем тесты перед рефакторингом
4. **Обратная совместимость** - старый API работает во время переходного периода

// src/core/EventBus.js
export class EventBus {
    constructor() {
        this.listeners = new Map();
        this.onceListeners = new Map();
    }
    
    on(event, callback) {
        if (!this.listeners.has(event)) {
            this.listeners.set(event, []);
        }
        this.listeners.get(event).push(callback);
        return () => this.off(event, callback); // Возвращаем функцию отписки
    }
    
    off(event, callback) {
        if (this.listeners.has(event)) {
            const callbacks = this.listeners.get(event);
            const index = callbacks.indexOf(callback);
            if (index > -1) {
                callbacks.splice(index, 1);
            }
        }
    }
    
    emit(event, data) {
        if (this.listeners.has(event)) {
            this.listeners.get(event).forEach(callback => {
                try {
                    callback(data);
                } catch (error) {
                    console.error(`Error in event handler for ${event}:`, error);
                }
            });
        }
        
        if (this.onceListeners.has(event)) {
            this.onceListeners.get(event).forEach(callback => callback(data));
            this.onceListeners.delete(event);
        }
    }
    
    once(event, callback) {
        if (!this.onceListeners.has(event)) {
            this.onceListeners.set(event, []);
        }
        this.onceListeners.get(event).push(callback);
    }
}
```

// src/core/GameStateManager.js
export class GameStateManager {
    constructor(eventBus) {
        this.eventBus = eventBus;
        this.states = {
            LOADING: 'loading',
            MENU: 'menu',
            PLAYING: 'playing',
            PAUSED: 'paused',
            GAME_OVER: 'game_over'
        };
        this.currentState = this.states.LOADING;
        this.stateData = {};
        this.stateHistory = [];
    }
    
    setState(newState, data = {}) {
        if (!Object.values(this.states).includes(newState)) {
            throw new Error(`Invalid state: ${newState}`);
        }
        
        const oldState = this.currentState;
        this.stateHistory.push({ state: oldState, timestamp: Date.now() });
        
        this.currentState = newState;
        this.stateData = { ...this.stateData, ...data };
        
        this.eventBus.emit('state_changed', {
            from: oldState,
            to: newState,
            data: this.stateData
        });
    }
    
    getState() {
        return this.currentState;
    }
    
    getStateData() {
        return this.stateData;
    }
    
    isState(state) {
        return this.currentState === state;
    }
}

export class Renderer {
    constructor(options = {}) {
        this.width = options.width || window.innerWidth;
        this.height = options.height || window.innerHeight;
        
        this.scene = new THREE.Scene();
        this.camera = new THREE.PerspectiveCamera(75, this.width / this.height, 0.1, 1000);
        this.webglRenderer = new THREE.WebGLRenderer({ 
            antialias: options.antialias !== false 
        });
        
        this.setupRenderer();
        this.setupLighting();
    }
    
    setupRenderer() {
        this.webglRenderer.setSize(this.width, this.height);
        this.webglRenderer.setPixelRatio(Math.min(window.devicePixelRatio, 1.5));
        this.webglRenderer.setClearColor(0x020502);
        this.webglRenderer.shadowMap.enabled = true;
        this.webglRenderer.shadowMap.type = THREE.PCFShadowMap;
        this.webglRenderer.powerPreference = "high-performance";
    }
    
    setupLighting() {
        // Базовое освещение
        this.ambientLight = new THREE.AmbientLight(0x404040, 0.3);
        this.scene.add(this.ambientLight);
    }
    
    render() {
        this.webglRenderer.render(this.scene, this.camera);
    }
    
    resize(width, height) {
        this.width = width;
        this.height = height;
        this.camera.aspect = width / height;
        this.camera.updateProjectionMatrix();
        this.webglRenderer.setSize(width, height);
    }
}

// src/core/ResourceManager.js
export class ResourceManager {
    constructor() {
        this.textures = new Map();
        this.sounds = new Map();
        this.models = new Map();
        this.loadingPromises = new Map();
        this.loadedCount = 0;
        this.totalCount = 0;
    }
    
    async loadTexture(path, options = {}) {
        if (this.textures.has(path)) {
            return this.textures.get(path);
        }
        
        if (this.loadingPromises.has(path)) {
            return this.loadingPromises.get(path);
        }
        
        const promise = new Promise((resolve, reject) => {
            const loader = new THREE.TextureLoader();
            loader.load(path, 
                (texture) => {
                    if (options.repeat) {
                        texture.wrapS = THREE.RepeatWrapping;
                        texture.wrapT = THREE.RepeatWrapping;
                        texture.repeat.set(...options.repeat);
                    }
                    this.textures.set(path, texture);
                    this.loadedCount++;
                    resolve(texture);
                },
                (progress) => {
                    // Прогресс загрузки
                    console.log(`Loading texture ${path}: ${progress.loaded / progress.total * 100}%`);
                },
                (error) => {
                    console.error(`Failed to load texture ${path}:`, error);
                    reject(error);
                }
            );
        });
        
        this.loadingPromises.set(path, promise);
        return promise;
    }
    
    async loadSound(path) {
        if (this.sounds.has(path)) {
            return this.sounds.get(path);
        }
        
        if (this.loadingPromises.has(path)) {
            return this.loadingPromises.get(path);
        }
        
        const promise = new Promise((resolve, reject) => {
            const loader = new THREE.AudioLoader();
            loader.load(path, 
                (buffer) => {
                    this.sounds.set(path, buffer);
                    this.loadedCount++;
                    resolve(buffer);
                },
                undefined,
                (error) => reject(error)
            );
        });
        
        this.loadingPromises.set(path, promise);
        return promise;
    }
    
    getLoadProgress() {
        return {
            loaded: this.loadedCount,
            total: this.totalCount,
            percentage: this.totalCount > 0 ? (this.loadedCount / this.totalCount) * 100 : 0
        };
    }
}
```

// src/physics/PhysicsSystem.js
export class PhysicsSystem {
    constructor() {
        this.gravity = -12.0;
        this.playerRadius = 0.35;
        this.collisionThreshold = 0.01;
    }
    
    checkPlayerCollision(position, mazeGrid, mazeSize) {
        const { x, z } = position;
        const r = this.playerRadius;
        
        const minX = Math.floor(x - r + 0.5);
        const maxX = Math.floor(x + r + 0.5);
        const minZ = Math.floor(z - r + 0.5);
        const maxZ = Math.floor(z + r + 0.5);
        
        for (let gy = minZ; gy <= maxZ; gy++) {
            for (let gx = minX; gx <= maxX; gx++) {
                if (gx < 0 || gx >= mazeSize || gy < 0 || gy >= mazeSize) {
                    return true;
                }
                
                if (mazeGrid[gy][gx] === 1) {
                    const testX = Math.max(gx - 0.5, Math.min(x, gx + 0.5));
                    const testZ = Math.max(gy - 0.5, Math.min(z, gy + 0.5));
                    const distX = x - testX;
                    const distZ = z - testZ;
                    
                    if ((distX * distX + distZ * distZ) <= r * r) {
                        return true;
                    }
                }
            }
        }
        return false;
    }
    
    applyGravity(velocity, deltaTime) {
        velocity.y += this.gravity * deltaTime;
    }
    
    checkJump(velocity, isGrounded) {
        if (isGrounded && velocity.y <= 0) {
            return 3.2; // Jump force
        }
        return 0;
    }
}

export class AudioManager {
    constructor(resourceManager, eventBus) {
        this.resourceManager = resourceManager;
        this.eventBus = eventBus;
        this.listener = new THREE.AudioListener();
        this.sounds = new Map();
        this.masterVolume = 1.0;
        this.isMuted = false;
        
        this.setupEventListeners();
    }
    
    setupEventListeners() {
        this.eventBus.on('monster_spawned', () => this.playMonsterSound());
        this.eventBus.on('key_collected', () => this.playCollectSound());
        this.eventBus.on('player_jump', () => this.playJumpSound());
    }
    
    async init() {
        // Предзагрузка звуков
        const soundPaths = [
            '/sounds/monster_ambient.mp3',
            '/sounds/scream1.mp3',
            '/sounds/light.mp3'
        ];
        
        for (const path of soundPaths) {
            await this.resourceManager.loadSound(path);
        }
    }
    
    playSound(soundName, options = {}) {
        if (this.isMuted) return;
        
        const buffer = this.resourceManager.sounds.get(soundName);
        if (!buffer) return;
        
        const sound = new THREE.Audio(this.listener);
        sound.setBuffer(buffer);
        sound.setVolume((options.volume || 1.0) * this.masterVolume);
        sound.setLoop(options.loop || false);
        
        if (options.position) {
            sound.position.copy(options.position);
        }
        
        sound.play();
        return sound;
    }
}

        this.eventBus.on('timer_updated', (data) => {
            this.updateTimer(data.time);

        switch (to) {
            case 'menu':
                this.showMenu();
                break;
            case 'playing':
                this.showHUD();
                break;
            case 'game_over':
                this.showGameOver();
                break;
        }

        this.hideAll();
        this.elements.get('menu').style.display = 'block';
    }
    
    showHUD() {
        this.hideAll();
        this.elements.get('hud').style.display = 'flex';
    }
    
    hideAll() {
        this.elements.forEach(element => {
            element.style.display = 'none';
        });
    }
}

// src/core/ObjectPool.js
export class ObjectPool {
    constructor(createFn, resetFn, initialSize = 10, maxSize = 100) {
        this.createFn = createFn;
        this.resetFn = resetFn;
        this.pool = [];
        this.active = new Set();
        this.maxSize = maxSize;
        
        // Предсоздание объектов
        for (let i = 0; i < initialSize; i++) {
            this.pool.push(this.createFn());
        }
    }
    
    acquire() {
        let obj;
        if (this.pool.length > 0) {
            obj = this.pool.pop();
        } else {
            obj = this.createFn();
        }
        
        this.active.add(obj);
        return obj;
    }
    
    release(obj) {
        if (!this.active.has(obj)) return;
        
        this.active.delete(obj);
        this.resetFn(obj);
        
        if (this.pool.length < this.maxSize) {
            this.pool.push(obj);
        }
    }
    
    releaseAll() {
        this.active.forEach(obj => this.release(obj));
    }
    
    getStats() {
        return {
            poolSize: this.pool.length,
            activeCount: this.active.size,
            totalCreated: this.pool.length + this.active.size
        };
    }
}

// src/renderer/LODManager.js
export class LODManager {
    constructor() {
        this.lodObjects = new Map();
        this.camera = null;
    }
    
    setCamera(camera) {
        this.camera = camera;
    }
    
    addLODObject(object, levels) {
        // levels: [{ distance: 10, object: highDetail }, { distance: 50, object: lowDetail }]
        this.lodObjects.set(object.uuid, { object, levels });
    }
    
    update() {
        if (!this.camera) return;
        
        this.lodObjects.forEach(({ object, levels }) => {
            const distance = this.camera.position.distanceTo(object.position);
            
            // Находим подходящий уровень детализации
            let selectedLevel = levels[levels.length - 1]; // Самый низкий по умолчанию
            for (const level of levels) {
                if (distance < level.distance) {
                    selectedLevel = level;
                    break;
                }
            }
            
            // Показываем только выбранный уровень
            levels.forEach(level => {
                level.object.visible = (level === selectedLevel);
            });
        });
    }
}

// src/ecs/System.js
export class System {
    constructor(requiredComponents = []) {
        this.requiredComponents = requiredComponents;
        this.entities = new Set();
    }
    
    addEntity(entity) {
        const hasAllComponents = this.requiredComponents.every(
            component => entity.hasComponent(component)
        );
        
        if (hasAllComponents) {
            this.entities.add(entity);
            this.onEntityAdded(entity);
        }
    }
    
    removeEntity(entity) {
        if (this.entities.has(entity)) {
            this.entities.delete(entity);
            this.onEntityRemoved(entity);
        }
    }
    
    update(deltaTime) {
        this.entities.forEach(entity => {
            this.updateEntity(entity, deltaTime);
        });
    }
    
    // Методы для переопределения
    onEntityAdded(entity) {}
    onEntityRemoved(entity) {}
    updateEntity(entity, deltaTime) {}
}

export class World {
    constructor() {
        this.entities = new Map();
        this.systems = new Map();
        this.entityIdCounter = 0;
    }

        const id = ++this.entityIdCounter;

        this.entities.set(id, entity);
        return entity;
    }

        this.systems.set(name, system);
        return this;
    }
    
    update(deltaTime) {
        this.systems.forEach(system => {
            system.update(deltaTime);
        });
    }
}
